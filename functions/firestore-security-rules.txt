rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // SOS Alerts - mobile app reads via Firestore SDK (volunteers are not Firebase Auth users)
    // App filters by district in query; only Cloud Functions write
    match /sos_alerts/{document} {
      allow read: if true;
      allow write: if false;
    }
    
    // Admins collection - authenticated users only
    match /admins/{email} {
      // Check if user has superadmin custom claim (set by Cloud Functions)
      function isSuperAdmin() {
        return request.auth != null && 
               request.auth.token.get('superadmin', false) == true;
      }
      
      // Allow super admins to read/write all admin documents
      allow read, write: if isSuperAdmin();
      
      // Allow admins to read their own document only
      allow read: if request.auth != null && request.auth.token.email == email;
    }
    
    // Blocked users - deny direct access (only via Cloud Functions)
    match /blocked_users/{document} {
      allow read, write: if false;
    }
  }
}
